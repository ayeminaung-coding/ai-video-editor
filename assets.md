## Usage Examples

### Using Icons in React
```typescript
// src/components/Icon.tsx
import React from 'react';

interface IconProps {
  name: string;
  size?: number;
  className?: string;
}

const Icon: React.FC<IconProps> = ({ name, size = 24, className = '' }) => {
  const icons: Record<string, string> = {
    upload: '/icons/upload.svg',
    play: '/icons/play.svg',
    pause: '/icons/pause.svg',
    edit: '/icons/edit.svg',
    export: '/icons/export.svg',
    settings: '/icons/settings.svg',
    user: '/icons/user.svg',
    delete: '/icons/delete.svg',
    download: '/icons/download.svg',
    video: '/icons/video.svg',
    music: '/icons/music.svg',
    text: '/icons/text.svg',
    speed: '/icons/speed.svg',
    volume: '/icons/volume.svg',
    brightness: '/icons/brightness.svg',
    contrast: '/icons/contrast.svg',
    trim: '/icons/trim.svg',
    dashboard: '/icons/dashboard.svg',
  };

  return (
    <img
      src={icons[name] || icons.upload}
      alt={name}
      width={size}
      height={size}
      className={className}
    />
  );
};

export default Icon;
```

### Using Icons with SVG Import
```typescript
// src/components/Icon.tsx
import React from 'react';
import UploadIcon from '../public/icons/upload.svg';
import PlayIcon from '../public/icons/play.svg';
import PauseIcon from '../public/icons/pause.svg';

interface IconProps {
  name: string;
  size?: number;
  className?: string;
}

const Icon: React.FC<IconProps> = ({ name, size = 24, className = '' }) => {
  const icons: Record<string, React.FC> = {
    upload: UploadIcon,
    play: PlayIcon,
    pause: PauseIcon,
  };

  const IconComponent = icons[name] || icons.upload;

  return (
    <IconComponent
      width={size}
      height={size}
      className={className}
    />
  );
};

export default Icon;
```
